🧩 调度器核心代码结构 
class Scheduler:
    """分组调度器"""

    def __init__(self):
        self.groups: Dict[str, SourceGroup] = {}
        self.task_queue: RedisQueue
        self.runner: TaskRunner
        self.last_collect_times: Dict[str, datetime] = {}

    async def run(self):
        """主循环 - 每分钟扫描"""
        while True:
            for group in self.groups.values():
                if self.should_trigger(group):
                    tasks = self.create_tasks(group)
                    await self.task_queue.push(tasks)
            await asyncio.sleep(60)  # 每分钟检查一次

    def should_trigger(self, group: SourceGroup) -> bool:
        """判断是否需要触发采集"""
        last_time = self.last_collect_times.get(group.id)
        if not last_time:
            return True
        elapsed = datetime.now() - last_time
        return elapsed >= timedelta(minutes=group.collect_interval)

    def create_tasks(self, group: SourceGroup) -> List[Task]:
        """为分组内所有数据源创建采集任务"""
        return [
            Task(source_id=source.id, priority=source.priority)
            for source in group.sources
            if source.is_active and source.status != 'paused'
        ]

class TaskRunner:
    """任务执行器"""

    async def run(self):
        """消费队列并执行"""
        while True:
            task = await self.task_queue.pop()
            await self.execute_task(task)

    async def execute_task(self, task: Task):
        """执行单个采集任务"""
        source = await self.get_source(task.source_id)
        collector = self.get_collector(source)
        result = await collector.collect(source, self.anti_crawl)
        await self.store_result(result, source)
        await self.log_execution(task, result)
📝 状态管理
数据源状态流转：
active → error → paused → active
调度器状态监控：
  • 任务队列长度
  • 当前执行任务数
  • 成功率统计
  • 平均采集耗时

调度特点： 分组统一频率管理，组内数据源按优先级排序。热榜组高频采集追踪变化，社交账号/短视频组低频采集全量去重。
